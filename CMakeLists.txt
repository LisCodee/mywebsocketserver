set(unit_test
    test/asynclog_unittest.cpp
    test/socket_unittest.cpp
    test/time_unittest.cpp
    test/_main.cpp
    test/threadpool_unittest.cpp
    )

    base/AsyncLog.h
    base/AsyncLog.cpp
    base/time.h
    base/time.cpp
    base/threadpool.h
    base/threadpool.cpp
    )
set(net 
    net/net.h
    net/net.cpp)
set(base )
# set(src src/main.cpp)

enable_testing()

    /usr/local/lib/libgtest.a
    /usr/local/lib/libgtest_main.a
)
